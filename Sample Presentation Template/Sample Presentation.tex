\defbeamertemplate{section in toc}{green circles}{%
  \leavevmode%
  \begin{beamercolorbox}[sep=-0.5pt,left]{section in toc}%
    \tikz[baseline=(tocprefix.base)]{
      \node[
        circle,
%        fill=green!70!black, % adjust to taste
	fill = Pantone633,
        inner sep=2pt,
        text=white
      ] (tocprefix) {\small\inserttocsectionnumber};
    }%
    \hspace{0.6em}\inserttocsection\par%
  \end{beamercolorbox}%
}
